<script>
export default {
  data () {

  methods: {
    accept () {
      let loader = this.$loading.show({
        container: this.fullPage ? null : this.$refs.formContainer,
        canCancel: false,
        loader: 'bars',
        color: '#0090FF',
        opacity: 0.1
      })
      setTimeout(() => {
        loader.hide()
        this.showNotif()
        this.$toast.open({
          message: 'Invitation accepted',
          type: 'success',
          position: 'bottom'
        })
      }, 1000)
    },
    decline () {
      let loader = this.$loading.show({
        container: this.fullPage ? null : this.$refs.formContainer,
        canCancel: false,
        loader: 'bars',
        color: '#0090FF',
        opacity: 0.1
      })
      setTimeout(() => {
        loader.hide()
        this.showNotif()
        this.$toast.open({
          message: 'Invitation declined',
          type: 'default',
          position: 'bottom'
        })
      }, 1000)
    },
    showNotif () {
      this.notification = !this.notification
    }
  }
